export const phoneNumberMask = (Form) => {
  const { phoneNumber } = Form;
  let newForm = {};
  if (phoneNumber == null || phoneNumber == '') {
    newForm = Object.assign({}, Form, {
      phoneNumber: '+7(___)___-__-__'
    });
  }
  return newForm;
};
